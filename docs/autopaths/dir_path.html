 17if os.name == "posix": sep = "/"
 18if os.name == "nt":    sep = "\\"

 21class DirectoryPath(autopaths.base_path.BasePath):
 22    """
 23    Holds a string pointing to a directory path and adds methods to interact
 24    with directories.
 25    """
 26
 27    def __len__(self): return len(list(self.flat_contents))
 28
 29    def __iter__(self): return self.flat_contents
 30
 31    def __contains__(self, item):
 32        return item in [x.name for x in self.flat_contents]
 33
 34    def __getitem__(self, item):
 35        for path in self.flat_contents:
 36            if path.name == item:
 37                return path
 38        raise KeyError("Couldn't find '%s' in '%s'" % (item, self.path))
 39
 40    #------------------------------- Properties ------------------------------#
 41    @property
 42    def p(self):
 43        if not hasattr(self, 'all_paths'):
 44            raise Exception("You need to define 'all_paths' to use this"
 45                            " function")
 46        return autopaths.auto_paths.AutoPaths(self.path, self.all_paths)
 47
 48    @property
 49    def name(self):
 50        """Just the directory name."""
 51        return os.path.basename(os.path.dirname(self.path))
 52
 53    @property
 54    def prefix_path(self):
 55        """The full path without the extension."""
 56        return os.path.splitext(self.path)[0].rstrip(sep)
 57
 58    @property
 59    def absolute_path(self):
 60        """The absolute path starting with a `/`."""
 61        return os.path.abspath(self.path) + sep
 62
 63    @property
 64    def directory(self):
 65        """The full path of the directory containing this one (parent dir)."""
 66        # The built-in function #
 67        directory = os.path.dirname(os.path.dirname(self.path))
 68        # Maybe we need to go the absolute path way #
 69        if not directory:
 70            directory = os.path.dirname(os.path.dirname(self.absolute_path))
 71        # Return #
 72        return autopaths.dir_path.DirectoryPath(directory)
 73
 74    @property
 75    def sub_directory(self):
 76        """
 77        The path to the directory that is contained inside this directory.
 78        Assuming that the current directory has only one child.
 79        """
 80        # Check there is only one #
 81        assert len(self.flat_directories) == 1
 82        # Return #
 83        return self.flat_directories[0]
 84
 85    @property
 86    def empty(self):
 87        """
 88        Return True if the directory contains no files or other directories.
 89        Will ignore items named `.DS_Store` and consider the directory empty.
 90        """
 91        for entry in self.flat_contents:
 92            if entry.name != '.DS_Store': break
 93        else: return True
 94        return False
 95
 96    @property
 97    def size(self):
 98        """The total size in bytes of all file contents."""
 99        total = sum(f.count_bytes for f in self.files)
100        return autopaths.file_size.FileSize(total)
101
102    #-------------------------- Recursive contents ---------------------------#
103    @property
104    def contents(self):
105        """The files and directories in this directory, recursively."""
106        for root, dirs, files in os.walk(self.path, topdown=False):
107            for d in dirs:
108                yield DirectoryPath(os.path.join(root, d))
109            for f in files:
110                yield autopaths.file_path.FilePath(os.path.join(root, f))
111
112    @property
113    def files(self):
114        """The files in this directory, recursively."""
115        for root, dirs, files in os.walk(self.path, topdown=False):
116            for f in files:
117                yield autopaths.file_path.FilePath(os.path.join(root, f))
118
119    @property
120    def directories(self):
121        """The directories in this directory, recursively."""
122        for root, dirs, files in os.walk(self.path, topdown=False):
123            for d in dirs:
124                yield DirectoryPath(os.path.join(root, d))
125
126    #----------------------------- Flat contents -----------------------------#
127    @property
128    def flat_contents(self):
129        """The files and directories in this directory non-recursively."""
130        for root, dirs, files in os.walk(self.path):
131            for d in dirs:
132                yield DirectoryPath(os.path.join(root, d))
133            for f in files:
134                yield autopaths.file_path.FilePath(os.path.join(root, f))
135            break
136
137    @property
138    def flat_files(self):
139        """
140        The files in this directory non-recursively, and sorted.
141        #TODO: check for permission denied in directory.
142        """
143        for root, dirs, files in os.walk(self.path):
144            result = [autopaths.file_path.FilePath(os.path.join(root, f))
145                      for f in files]
146            break
147        else:
148            result = []
149        result.sort(key=lambda x: autopaths.common.natural_sort(x.path))
150        return result
151
152    @property
153    def flat_directories(self):
154        """The directories in this directory non-recursively, and sorted."""
155        for root, dirs, files in os.walk(self.path):
156            result = [DirectoryPath(os.path.join(root, d)) for d in dirs]
157            break
158        else:
159            result = []
160        result.sort(key=lambda x: autopaths.common.natural_sort(x.path))
161        return result
162
163    #------------------------------- Methods ---------------------------------#
164    def must_exist(self):
165        """Raise an exception if the directory doesn't exist."""
166        if not os.path.isdir(self.path):
167            msg = "The directory path '%s' does not exist."
168            raise Exception(msg % self.path)
169
170    def remove(self, safe=True):
171        if not self.exists: return False
172        if self.is_symlink: return self.remove_when_symlink()
173        shutil.rmtree(self.path, ignore_errors=safe)
174        return True
175
176    def remove_when_symlink(self):
177        if not self.exists: return False
178        os.remove(self.path.rstrip(sep))
179        return True
180
181    def create(self, safe=False, inherit=True):
182        # Create it #
183        if not safe:
184            os.makedirs(self.path)
185            if inherit: os.chmod(self.path, self.directory.permissions.number)
186        if safe:
187            try:
188                os.makedirs(self.path)
189                if inherit:
190                    os.chmod(self.path, self.directory.permissions.number)
191            except OSError: pass
192
193    def create_if_not_exists(self):
194        if not self.exists: self.create()
195
196    def move_to(self, path):
197        """Move the directory."""
198        # Parse #
199        path = DirectoryPath(path)
200        # Check the destination doesn't exist already #
201        assert not path.exists
202        # However, the parent directory must exist #
203        path.directory.create_if_not_exists()
204        # Move #
205        shutil.move(self.path, path)
206        # Update the internal link #
207        self.path = path
208
209    def copy(self, path):
210        assert not os.path.exists(path)
211        shutil.copytree(str(self.path), str(path))
212
213    def glob(self, pattern):
214        """Perform a glob search in this directory."""
215        files = glob.glob(self.path + pattern)
216        return list(map(autopaths.file_path.FilePath, files))
217
218    def find(self, pattern):
219        """Find a file in this directory."""
220        f = glob.glob(self.path + pattern)[0]
221        return autopaths.file_path.FilePath(f)
222
223    def unnest(self):
224        """
225        Move all contents (files and directories) of this directory to its
226        parent directory, and remove this directory.
227        """
228        for item in self.flat_contents: item.move_to(self.directory)
229        self.remove()
230
231    def remove_empty_dirs(self):
232        """
233        Find all empty directories within this directory recursively,
234        and remove them.
235        """
236        for directory in self.directories:
237            if directory.empty:
238                directory.remove()
239
240    #---------------------------- TAR compression ----------------------------#
241    def tar_to(self, path=None):
242        """Make a tared version of the directory at a given path."""
243        pass
244
245    def targz_to(self, path=None):
246        """Make a targzipped version of the directory at a given path."""
247        pass
248
249    #---------------------------- ZIP compression ----------------------------#
250    def zip_to(self, path=None):
251        """Make a zipped version of the directory at a given path."""
252        # Case where path is None #
253        if path is None: path = self.directory + self.name + ".zip"
254        else: path = autopaths.file_path.FilePath(path)
255        # The method adds .zip so let's remove it #
256        if path.endswith('.zip'): path_no_ext = path.prefix_path
257        else:                     path_no_ext = path
258        # Compress #
259        shutil.make_archive(path_no_ext, 'zip', self.path)
260        # Return #
261        return path_no_ext
