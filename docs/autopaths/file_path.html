253    def tail(self, num_lines=20, encoding='utf-8'):
254        """Yield the last few lines of the file."""
255        # Constant #
256        buffer_size = 1024
257        # Smart algorithm #
258        with open(self.path, 'rb') as handle:
259            handle.seek(0, 2)
260            num_bytes = handle.tell()
261            size      = num_lines + 1
262            block     = -1
263            data      = []
264            # Loop #
265            while size > 0 and num_bytes > 0:
266                if num_bytes - buffer_size > 0:
267                    # Seek back one whole buffer_size #
268                    handle.seek(block * buffer_size, os.SEEK_END)
269                    # Read buffer #
270                    data.insert(0, handle.read(buffer_size).decode(encoding))
271                else:
272                    # File too small, start from beginning #
273                    handle.seek(0,0)
274                    # Only read what was not read #
275                    data.insert(0, handle.read(num_bytes).decode(encoding))
276                lines_found = data[0].count('\n')
277                size       -= lines_found
278                num_bytes  -= buffer_size
279                block      -= 1
280            # Return #
281            for line in ''.join(data).splitlines()[-num_lines:]: yield line

314    def gzip_to(self, new_path=None, remove_orig=False, method='prll'):
315        """
316        Make a gzipped version of the file at a given path.
317        If the path already contains `.gz` and you want to compress inplace,
318        just specify `new_path=False`.
319        """
320        # Case where the path is not specified #
321        if new_path is None:
322            new_path = self.path + '.gz'
323        # In case we want to do it in place #
324        if new_path is False:
325            remove_orig = True
326            in_place = True
327            new_path = new_temp_file(prefix='gzip_to-')
328        # Do it the fast way or the slow way #
329        if method == 'ext':  self.gzip_external(new_path)
330        if method == 'pigz': self.gzip_pigz(new_path)
331        else:                self.gzip_internal(new_path)
332        # Move the temporary file back #
333        if in_place:
334            new_path.move_to(self.path, overwrite=True)
335        # Optionally remove the original uncompressed file #
336        elif remove_orig:
337            self.remove()
338        # Update the internal path #
339        else: self.path = new_path
340        # Return #
341        return self.path

360    def gzip_pigz(self, new_path):
361        """
362        Do the compression with multiple threads. Possible projects:
363        * https://github.com/madler/pigz (we choose this one)
364        * https://github.com/klauspost/pgzip
365        """
366        # Get the command #
367        import sh
368        pigz = sh.Command("pigz")
369        # Command line options #
370        options = {'keep': True}
371        # It will refuse to compress files ending in '.gz' #
372        if self.path.endswith('.gz'):
373            working_path = self + '.pigz'
374            shutil.move(self, working_path)
375        else:
376            working_path = self
377        # Run it #
378        pigz(working_path, **options)
379        # Restore the original file if the '.pigz' extension was added
380        # to it earlier.
381        if self.path.endswith('.gz'):
382            shutil.move(working_path, self)
383        # You can't specify the output path of pigz, it's always this #
384        out_path = working_path + '.gz'
385        # Move the file to the destination the user wants #
386        if out_path != new_path:
387            out_path.move_to(new_path, overwrite=True)

414    def unzip_to(self, path=None, inplace=False, single=True):
415        """
416        Unzip a standard zip file. Can specify the destination of the
417        uncompressed file, or just set inplace=True to delete the original.
418        """
419        # Parse the path #
420        path = autopaths.Path(path)
421        # Check #
422        assert zipfile.is_zipfile(self.path)
423        # Load #
424        z = zipfile.ZipFile(self.path)
425        if single or inplace: assert len(z.infolist()) == 1
426        # Single file #
427        if single:
428            member = z.infolist()[0]
429            tmpdir = tempfile.mkdtemp() + sep
430            z.extract(member, tmpdir)
431            z.close()
432            if inplace: shutil.move(tmpdir + member.filename,
433                                    self.directory + member.filename)
434            else:       shutil.move(tmpdir + member.filename,
435                                    path)
436        # Multifile - no security, dangerous - Will use CWD if dest is None!!
437        # If a file starts with an absolute path, will overwrite your files
438        # anywhere
439        if not single:
440            z.extractall(path)
441            return autopaths.dir_path.DirectoryPath(path)
442        # Return #
443        return FilePath(path)

254    def tail(self, num_lines=20, encoding='utf-8'):
255        """Yield the last few lines of the file."""
256        # Constant #
257        buffer_size = 1024
258        # Smart algorithm #
259        with open(self.path, 'rb') as handle:
260            handle.seek(0, 2)
261            num_bytes = handle.tell()
262            size      = num_lines + 1
263            block     = -1
264            data      = []
265            # Loop #
266            while size > 0 and num_bytes > 0:
267                if num_bytes - buffer_size > 0:
268                    # Seek back one whole buffer_size #
269                    handle.seek(block * buffer_size, os.SEEK_END)
270                    # Read buffer #
271                    data.insert(0, handle.read(buffer_size).decode(encoding))
272                else:
273                    # File too small, start from beginning #
274                    handle.seek(0,0)
275                    # Only read what was not read #
276                    data.insert(0, handle.read(num_bytes).decode(encoding))
277                lines_found = data[0].count('\n')
278                size       -= lines_found
279                num_bytes  -= buffer_size
280                block      -= 1
281            # Return #
282            for line in ''.join(data).splitlines()[-num_lines:]: yield line
